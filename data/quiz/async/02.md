function getColor ( num ) {
    var colors = [ "blue", "green", "yellow", "orange", "red" ]
    return new Promise ( function ( resolve, reject ) {
        setTimeout ( () => {
            num < 0 || num > colors.length - 1 ?
                   reject ( "Range error" ) :
                   resolve ( colors [ num ] )
        }, 1000 )
    })
}
async function getAllColors () {

        )

            width: 50px;
            height: 50px;
            background-color: ${await getColor ( x )};
        `
    }
}
