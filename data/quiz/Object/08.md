function User ( name ) {
    var userName = name || "User"
    var postList = []
    Object.defineProperties ( this, {
        name: {
            get () {
                return userName
            },
            set ( newName ) {
                userName = newName
            }
        },
        posts: {
            get () {
                return JSON.parse (
                    JSON.stringify ( postList )
                )
            },
            set ( postData ) {
                postData && postData.header && postData.text ?
                    postList.push ({
                        title: postData.header || "...",
                        text: postData.text || "...empty",
                    }) : null
            }
        }
    })
}
