function* parentElements ( collection ) {
    let currentNum = 0
    let result = []
    while ( !!collection [ currentNum ] ) {
        result.push ( collection [ currentNum++ ].parentNode )
    }
    yield result
}

let collection = document.getElementsByTagName ( "p" )
