____quizJS____
var sample = [ "abc", false, 5, 1, true, 10, undefined, null ]

function buildArrayOfNumbers ( oldArray ) {
    var newArray = []
    for ( i = 0;   i < oldArray.length;   i++ ) {
        if ( typeof oldArray [ i ]  === "number" ) {
            newArray.push ( oldArray [ i ] )
        }
    }
    return newArray
}

console.log (
    buildArrayOfNumbers ( sample )
)
