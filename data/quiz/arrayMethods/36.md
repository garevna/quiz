function reverseArray ( arr ) {
    var res = [ ]
    for ( var j = arr.length - 1;   j >= 0;   j-- ) {
        res.push ( arr [ j ] )
    }
    return res
}
var meat = [ "курица", "шашлык", "рыба" ]
