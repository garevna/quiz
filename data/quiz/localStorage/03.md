var email = document.querySelector ( '#email' )
var passw = document.querySelector ( '#password' )

function saveData() {
    if ( !email.value || !passw.value ) return
    var user = {
        name: email.value,
        pass: Sha256.hash ( passw.value )
    }
    localStorage.setItem (
        "user",
        JSON.stringify ( user )
    )
}
