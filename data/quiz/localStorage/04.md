____quizJS____
let collection = document.querySelectorAll (
        "input[type='checkbox']"
)
var fruits = []

function selectFruit ( event ) {
    if ( this.checked ) fruits.push ( this.value )
    else fruits.splice ( fruits.indexOf ( this.value ), 1 )
}

function saveChoice () {
    localStorage.setItem (
        "selectedFruits",
        JSON.stringify ( fruits )
    )
}
